# S6

Sparrow6 command line client and plugin manager.

# Usage

help:

    s6 --help

actions:

    s6 $action|$options $thing

update index:

    s6 --index-update

upload plugin:

    s6 --upload

install plugin:

    s6 --install $plugin

    s6 --install --force $plugin # reinstall, even though if higher version is here

options:

  --debug   # enable debug mode
